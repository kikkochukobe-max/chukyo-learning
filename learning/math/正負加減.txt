<script>
"use strict";

const cfg={termCount:'2',maxTerms:3,brackets:'none',numType:'integer'};
let score={total:0,correct:0,wrong:0},probNum=0,curAns=null,answered=false,stgOpen=true;

let iSign='+', numVal='', denVal='', isFrac=false, activeField='num';

function gcd(a,b){a=Math.abs(Math.round(a));b=Math.abs(Math.round(b));while(b){const t=b;b=a%b;a=t;}return a||1;}

class Frac{
  constructor(n,d=1){n=Math.round(n);d=Math.round(d);if(!d)throw Error('0');if(d<0){n=-n;d=-d;}const g=gcd(Math.abs(n),d);this.n=n/g;this.d=d/g;}
  add(o){return new Frac(this.n*o.d+o.n*this.d,this.d*o.d);}
  abs(){return new Frac(Math.abs(this.n),this.d);}
  isInt(){return this.d===1;}
  isNeg(){return this.n<0;}
  eq(o){return this.n===o.n&&this.d===o.d;}
  isDecimal2(){let d=this.d;while(d%2===0)d/=2;while(d%5===0)d/=5;if(d!==1)return false;const dec=(Math.abs(this.n)/this.d).toFixed(10).replace(/\.?0+$/,'').split('.')[1]||'';return dec.length<=2;}
  toDecStr(){const v=this.n/this.d;for(let p=0;p<=2;p++){if(Math.abs(parseFloat(v.toFixed(p))-v)<1e-10)return parseFloat(v.toFixed(p)).toString();}return parseFloat(v.toFixed(2)).toString();}
}

function ri(a,b){return a+Math.floor(Math.random()*(b-a+1));}
function genInt(){const v=ri(1,15);return{value:new Frac(Math.random()<.5?v:-v),type:'int'};}
function genFrac(){
  for(let i=0;i<100;i++){const d=ri(2,10),n=ri(1,Math.min(20,d+8)),g=gcd(n,d),nd=n/g,dd=d/g;if(dd>1&&dd<=15&&nd<=20)return{value:new Frac(Math.random()<.5?nd:-nd,dd),type:'frac'};}
  return genInt();
}

const DPOOL=(()=>{
  const p=[];
  for(let i=1;i<=19;i++){
    if(i===10)continue;
    const g=gcd(i,10);
    p.push({n:i/g,d:10/g,s:(i/10).toString()});
  }
  [[1,4,0.25],[3,4,0.75],[5,4,1.25],[7,4,1.75],[9,4,2.25],[11,4,2.75],
   [1,2,0.5],[3,2,1.5],[5,2,2.5]].forEach(([n,d,v])=>p.push({n,d,s:v.toString()}));
  return p;
})();

function genDecimal(){const p=DPOOL[Math.floor(Math.random()*DPOOL.length)];const s=Math.random()<.5?1:-1;return{value:new Frac(s*p.n,p.d),type:'dec',displayAbs:p.s};}
function genTerm(){if(cfg.numType==='integer')return genInt();const r=Math.random();if(r<.4)return genInt();if(r<.72)return genFrac();return genDecimal();}

function genProblem(){
  const nT=cfg.termCount==='2'?2:ri(3,cfg.maxTerms);
  for(let a=0;a<300;a++){
    const terms=Array.from({length:nT},()=>genTerm());
    if(!terms.every(t=>{const f=t.value;return f.isInt()?Math.abs(f.n)<=20:(f.d<=15&&Math.abs(f.n)<=20);}))continue;
    let ans=terms[0].value;
    for(let i=1;i<terms.length;i++)ans=ans.add(terms[i].value);
    if(!ans.isInt()&&(ans.d>15||Math.abs(ans.n)>20))continue;
    return{terms,answer:ans};
  }
  const n=cfg.termCount==='2'?2:3;
  const terms=Array.from({length:n},()=>genInt());
  let ans=terms[0].value;
  for(let i=1;i<terms.length;i++)ans=ans.add(terms[i].value);
  return{terms,answer:ans};
}

function assignDisplay(n){
  if(cfg.brackets==='none'){
    return{brks:Array(n).fill(false),ops:Array(n).fill('+')};
  }
  if(cfg.brackets==='all'){
    const brks=Array(n).fill(true);
    const ops=['+',...Array.from({length:n-1},()=>Math.random()<.5?'+':'-')];
    return{brks,ops};
  }
  const brks=Array.from({length:n},()=>Math.random()<.5);
  if(n>=2){if(brks.every(x=>x))brks[ri(0,n-1)]=false;if(brks.every(x=>!x))brks[ri(0,n-1)]=true;}
  const ops=['+',...Array.from({length:n-1},(_,i)=>brks[i+1]?(Math.random()<.5?'+':'-'):'+')];
  return{brks,ops};
}

function absHTML(t){
  const f=t.value.abs();
  if(t.type==='dec')return`<span>${t.displayAbs}</span>`;
  if(f.isInt())return`<span>${f.n}</span>`;
  return`<span class="f"><span class="fn">${f.n}</span><span class="fd">${f.d}</span></span>`;
}

function buildHTML(terms,brks,ops){
  let h='';
  for(let i=0;i<terms.length;i++){
    const t=terms[i],neg=t.value.isNeg(),av=absHTML(t),brk=brks[i],op=ops[i];
    if(brk){
      if(i===0){
        const sg=neg?'<span class="sgn">−</span>':'<span class="sgn">+</span>';
        h+=`<span class="brk">(</span>${sg}${av}<span class="brk">)</span>`;
      }else{
        const displayNeg=(op==='+')?neg:!neg;
        const sg=displayNeg?'<span class="sgn">−</span>':'<span class="sgn">+</span>';
        const opHTML=op==='+'?'＋':'−';
        h+=`<span class="op"> ${opHTML} </span><span class="brk">(</span>${sg}${av}<span class="brk">)</span>`;
      }
    }else{
      if(i===0)h+=`${neg?'<span class="op">−</span>':''}${av}`;
      else h+=`${neg?'<span class="op"> − </span>':'<span class="op"> ＋ </span>'}${av}`;
    }
  }
  return h;
}

function ansToHTML(f){
  if(f.isInt())return`<span style="font-family:'Fira Mono',monospace;font-size:1rem">${f.n}</span>`;
  if(f.isDecimal2())return`<span style="font-family:'Fira Mono',monospace;font-size:1rem">${f.toDecStr()}</span>`;
  const a=f.abs(),sign=f.isNeg()?'<span style="color:var(--red)">−</span>':'';
  return`${sign}<span class="f" style="font-size:.95rem"><span class="fn">${a.n}</span><span class="fd">${a.d}</span></span>`;
}

function updateInputUI(){
  document.getElementById('num-txt').textContent=numVal;
  document.getElementById('den-txt').textContent=denVal;
  document.getElementById('inp-num').classList.toggle('active',activeField==='num');
  document.getElementById('inp-den').classList.toggle('active',activeField==='den');
  const showDen=isFrac;
  document.getElementById('frac-div').style.display=showDen?'':'none';
  document.getElementById('inp-den').style.display=showDen?'':'none';
  document.getElementById('kbtn-dot').classList.toggle('k-disabled',isFrac);
  document.getElementById('kbtn-slash').classList.toggle('k-disabled',isFrac||numVal.includes('.'));
}

function setActive(field){
  if(answered)return;
  if(field==='den'&&!isFrac)return;
  activeField=field;
  updateInputUI();
}

function toggleSign(){
  if(answered)return;
  iSign=iSign==='+'?'-':'+';
  const btn=document.getElementById('sign-btn');
  btn.textContent=iSign==='+'?'＋':'−';
  btn.className='sign-toggle '+(iSign==='+'?'pos':'neg');
}

function kNum(ch){
  if(answered)return;
  if(activeField==='num'){
    if(numVal.length>=6)return;
    if(ch==='0'&&numVal===''&&!isFrac){numVal='0';updateInputUI();return;}
    numVal+=ch;
  }else{
    if(denVal.length>=2)return;
    if(ch==='0'&&denVal==='')return;
    denVal+=ch;
  }
  updateInputUI();
}

function kSlash(){
  if(answered||isFrac||numVal.includes('.'))return;
  isFrac=true;
  activeField='den';
  updateInputUI();
}

function kDot(){
  if(answered||isFrac||activeField==='den')return;
  if(numVal.includes('.'))return;
  if(numVal===''||numVal==='0')numVal='0';
  numVal+='.';
  updateInputUI();
}

function kBack(){
  if(answered)return;
  if(activeField==='den'){
    if(denVal.length>0){denVal=denVal.slice(0,-1);}
    else{isFrac=false;activeField='num';}
  }else{
    if(numVal.length>0)numVal=numVal.slice(0,-1);
  }
  updateInputUI();
}

function kClear(){
  if(answered)return;
  iSign='+';numVal='';denVal='';isFrac=false;activeField='num';
  const btn=document.getElementById('sign-btn');
  btn.textContent='＋';btn.className='sign-toggle pos';
  updateInputUI();
}

function getAnswerFrac(){
  if(!numVal||numVal==='.'||numVal==='0.')return null;
  const signMult=iSign==='-'?-1:1;
  if(isFrac){
    if(!denVal)return null;
    const n=parseInt(numVal,10),d=parseInt(denVal,10);
    if(isNaN(n)||isNaN(d)||d===0)return null;
    try{return new Frac(signMult*n,d);}catch(e){return null;}
  }
  const v=parseFloat(numVal);
  if(isNaN(v))return null;
  const m=numVal.match(/\.(\d+)$/);
  if(m){const mult=Math.pow(10,m[1].length);return new Frac(signMult*Math.round(v*mult),mult);}
  return new Frac(signMult*Math.round(v));
}

document.addEventListener('keydown',e=>{
  if(answered){if(e.key==='Enter')newProb();return;}
  if(e.key==='Enter'){checkAns();return;}
  if(e.key==='Backspace'){kBack();return;}
  if(e.key==='Delete'){kClear();return;}
  if(e.key==='-'||e.key==='−'){toggleSign();return;}
  if(e.key==='/'){kSlash();return;}
  if(e.key==='.'){kDot();return;}
  if(/^\d$/.test(e.key))kNum(e.key);
});

function updScore(){
  const rate=score.total>0?Math.round(score.correct/score.total*100):null;
  document.getElementById('sc-t').textContent=score.total;
  document.getElementById('sc-c').textContent=score.correct;
  document.getElementById('sc-w').textContent=score.wrong;
  document.getElementById('sc-r').textContent=rate!==null?`${rate}%`:'—';
  document.getElementById('hd-c').textContent=score.correct;
  document.getElementById('hd-t').textContent=score.total;
  document.getElementById('hd-r').textContent=rate!==null?`${rate}%`:'—';
}

function newProb(){
  const prob=genProblem();
  curAns=prob.answer;answered=false;probNum++;

  const{brks,ops}=assignDisplay(prob.terms.length);
  document.getElementById('math-disp').innerHTML=buildHTML(prob.terms,brks,ops);
  document.getElementById('prob-no').textContent=`問題 #${probNum}`;

  iSign='+';numVal='';denVal='';isFrac=false;activeField='num';
  document.getElementById('sign-btn').textContent='＋';
  document.getElementById('sign-btn').className='sign-toggle pos';
  document.getElementById('ans-area').classList.remove('locked');
  document.getElementById('inp-num').className='inp-field active';
  document.getElementById('inp-den').className='inp-field';
  document.getElementById('frac-div').className='frac-divider';
  updateInputUI();

  document.getElementById('fb').innerHTML='';
  document.getElementById('btn-chk').style.display='';
  document.getElementById('btn-nxt').style.display='none';
  document.getElementById('keypad').style.opacity='1';
  document.getElementById('keypad').style.pointerEvents='';
  document.getElementById('kpad-hint').style.display='';
}

function checkAns(){
  if(answered)return;
  const ua=getAnswerFrac();
  if(!ua){
    document.getElementById('fb').innerHTML='<span style="color:var(--amber)">⚠ 答えを入力してください</span>';
    return;
  }
  answered=true;score.total++;
  const ok=ua.eq(curAns);

  document.getElementById('ans-area').classList.add('locked');
  const numField=document.getElementById('inp-num');
  const denField=document.getElementById('inp-den');
  const divider=document.getElementById('frac-div');

  if(ok){
    score.correct++;
    numField.className='inp-field result-ok pulsing';
    if(isFrac){denField.className='inp-field result-ok';divider.className='frac-divider result-ok';}
    setTimeout(()=>numField.classList.remove('pulsing'),600);
    document.getElementById('fb').innerHTML='<span style="color:var(--green);font-size:1.05rem">✓ 正解！</span>';
  }else{
    score.wrong++;
    numField.className='inp-field result-ng shaking';
    if(isFrac){denField.className='inp-field result-ng';divider.className='frac-divider result-ng';}
    setTimeout(()=>numField.classList.remove('shaking'),400);
    document.getElementById('fb').innerHTML=
      `<span style="color:var(--red)">✗ 不正解</span><span style="color:var(--muted)">　正解：</span><span style="color:var(--green)">${ansToHTML(curAns)}</span>`;
  }

  document.getElementById('btn-chk').style.display='none';
  document.getElementById('btn-nxt').style.display='';
  document.getElementById('keypad').style.opacity='0.35';
  document.getElementById('keypad').style.pointerEvents='none';
  document.getElementById('kpad-hint').style.display='none';
  updScore();
}

function skipProb(){newProb();}

}
